import requests
from lxml import etree

class ListSpider(object):
def init(self):
self.TOTAL_PAGE = 10
self.BASE_URL = “http://quotes.toscrape.com"
self.headers = {
“User-Agent”: “Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/81.0.4044.129 Safari/537.36”
}

def crawl(self, index=1):
    url = self.BASE_URL + f"/page/{index}/"
    r = requests.get(url=url, headers=self.headers)
    # with open("quotes.html", "w", encoding="utf8") as file:
    #     file.write(r.content.decode("utf8"))
    return r.content.decode("utf8")

@staticmethod
def parser(html):
    # with open("quotes.html", "r", encoding="utf8") as file:
    #     html = file.read()
    element = etree.HTML(html)
    saying = element.xpath("//div[@class='quote']/span[@class='text']/text()")
    author = element.xpath("//div[@class='quote']/span[2]/small/text()")
    tags = element.xpath("//div[@class='tags']/a[@href]/text()")
    for i in range(len(saying)):
        message = {
            "名言": saying[i],
            "作者": author[i],
            "标签": tags,
            "status": 0
        }
        print(message)

def run(self):
    #         for i in range(self.TOTAL_PAGE):
    #             self.crawl(i)
    #             break
    html = self.crawl()
    self.parser(html)
